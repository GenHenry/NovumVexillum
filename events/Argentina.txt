	id = argentina.3
	title = argentina.3.t
	desc = argentina.3.d
	picture = GFX_report_event_president_resigns

	is_triggered_only = yes
	
	option = {
		name = argentina.3.a
		add_stability = -0.2
		
		retire_character = ARG_fernando_de_la_rua
		promote_character = ARG_federico_ramon_puerta

		set_politics = {
			ruling_party = conservative
			elections_allowed = yes
			long_name = ARG_partido_justicialista_long
			name = ARG_partido_justicialista
		}
		hidden_effect = {
			set_popularities = {
				nationalist = 4
				reactionary = 6
				conservative = 40
				social_liberal = 1
				social_democrat = 8
				progressive = 1
				democratic_socialist = 37
				communist = 3
			}
			remove_ideas = fourth_estate_social_democrat4
			remove_ideas = fourth_estate_social_democrat3
			remove_ideas = fourth_estate_social_democrat2
			remove_ideas = fourth_estate_social_democrat1
			remove_ideas = fourth_estate_conservative2
			add_ideas = fourth_estate_democratic_socialist4
			add_ideas = fourth_estate_conservative4
			country_event = {
				id = argentina.4
				days = 2
			}
		}	
	}
}
